# Streaming Guide — Implementation Plan

> **For Hermes:** Use subagent-driven-development skill to implement this plan task-by-task.

**Goal:** Build a modern streaming guide — a single grid-based interface showing what's on across broadcast TV and streaming platforms, starting with US listings.

**Architecture:** Python `http.server` backend with SQLite cache, vanilla HTML/CSS/JS frontend, grid-based UI. Tailscale-bound on port 8788. Data from TVmaze (broadcast, free) and TMDB (streaming availability, free API key).

**Tech Stack:** Python 3 (stdlib + sqlite3), HTML/CSS/JS (no build step), TVmaze API, TMDB API

---

## Data Sources

| Source | Cost | What it Provides | Rate Limits |
|---|---|---|---|
| **TVmaze** | Free | Broadcast schedules, show/episode data, network info, search | 20 calls/10s (no key needed) |
| **TMDB** | Free key | Streaming availability per country, show/movie metadata, posters | 50 req/s (free tier) |
| **Schedules Direct** | $25/yr | Full US/Canada broadcast grid (what Plex/Emby use) | Add in v2 |

TVmaze gives us "what's on now" for broadcast. TMDB gives us "where to stream it" for on-demand. Combined, we cover both sides.

---

## MVP Scope

- Grid view: channels × 2-hour time window
- Broadcast listings from TVmaze (US)
- Streaming availability badges from TMDB ("Netflix", "Prime", etc.)
- Show details on click (synopsis, network, streaming links)
- Search by show name
- No accounts, no favorites, no DVR. Just listings.

```python
#!/usr/bin/env python3
"""Streaming Guide — broadcast + streaming listings in a TV Guide grid."""
import json, sqlite3, os
from http.server import HTTPServer, SimpleHTTPRequestHandler
from pathlib import Path
from urllib.parse import urlparse, parse_qs

ROOT = Path(__file__).parent
STATIC = ROOT / "static"
DATA = ROOT / "data"
DATA.mkdir(exist_ok=True)

class Handler(SimpleHTTPRequestHandler):
    def __init__(self, *args, **kwargs):
        super().__init__(*args, directory=str(STATIC), **kwargs)

    def do_GET(self):
        path = urlparse(self.path).path
        if path == "/api/health":
            self.send_json({"status": "ok"})
        elif path.startswith("/api/"):
            self.send_json({"error": "not implemented"}, 404)
        else:
            super().do_GET()

    def send_json(self, data, status=200):
        self.send_response(status)
        self.send_header("Content-Type", "application/json")
        self.send_header("Cache-Control", "no-store")
        self.end_headers()
        self.wfile.write(json.dumps(data).encode())

if __name__ == "__main__":
    port = int(os.environ.get("PORT", 8788))
    server = HTTPServer(("0.0.0.0", port), Handler)
    print(f"Streaming Guide on :{port}")
    server.serve_forever()
```

## Task 2: SQLite cache layer

**Objective:** Create cache database and utility functions to avoid hammering APIs

**Files:**
- Modify: `~/agentos/streaming-guide/server.py`

**Step 1:** Add `init_db()` and `cache_get/set` functions

```python
import time

DB_PATH = DATA / "cache.db"

def init_db():
    db = sqlite3.connect(str(DB_PATH))
    db.execute("""
        CREATE TABLE IF NOT EXISTS cache (
            key TEXT PRIMARY KEY,
            value TEXT NOT NULL,
            expires_at REAL NOT NULL
        )
    """)
    db.commit()
    return db

def cache_get(db, key):
    row = db.execute(
        "SELECT value, expires_at FROM cache WHERE key = ?", (key,)
    ).fetchone()
    if row and row[1] > time.time():
        return json.loads(row[0])
    return None

def cache_set(db, key, value, ttl=300):
    db.execute(
        "INSERT OR REPLACE INTO cache (key, value, expires_at) VALUES (?, ?, ?)",
        (key, json.dumps(value), time.time() + ttl)
    )
    db.commit()

# Initialize DB at startup
init_db()
```

import urllib.request
import urllib.error

TVMAZE_BASE = "https://api.tvmaze.com"

def fetch_tvmaze(path, db):
    """Fetch from TVmaze API with cache."""
    key = f"tvmaze:{path}"
    cached = cache_get(db, key)
    if cached:
        return cached

    url = f"{TVMAZE_BASE}{path}"
    req = urllib.request.Request(url, headers={"User-Agent": "StreamingGuide/0.1"})
    try:
        with urllib.request.urlopen(req, timeout=10) as resp:
            data = json.loads(resp.read())
            cache_set(db, key, data, ttl=3600)  # 1 hour TTL for schedules
            return data
    except urllib.error.URLError as e:
        print(f"TVmaze error: {e}")
        return None

def get_us_schedule(db):
    """Get today's US broadcast schedule."""
    from datetime import date
    today = date.today().isoformat()
    return fetch_tvmaze(f"/schedule?country=US&date={today}", db) or []
```

def search_tmdb(query, db):
    """Search TMDB for a show by name."""
    key = f"tmdb:search:{query}"
    cached = cache_get(db, key)
    if cached:
        return cached
    if not TMDB_KEY:
        return None
    
    url = f"{TMDB_BASE}/search/tv?api_key={TMDB_KEY}&query={urllib.parse.quote(query)}"
    req = urllib.request.Request(url, headers={"User-Agent": "StreamingGuide/0.1"})
    try:
        with urllib.request.urlopen(req, timeout=10) as resp:
            data = json.loads(resp.read())
            cache_set(db, key, data, ttl=86400)  # 24h TTL for search
            return data
    except Exception as e:
        print(f"TMDB error: {e}")
        return None

def get_streaming_providers(tmdb_id, db):
    """Get streaming availability for a show (US)."""
    key = f"tmdb:providers:{tmdb_id}"
    cached = cache_get(db, key)
    if cached:
        return cached
    if not TMDB_KEY:
        return {}
    
    url = f"{TMDB_BASE}/tv/{tmdb_id}/watch/providers?api_key={TMDB_KEY}"
    req = urllib.request.Request(url, headers={"User-Agent": "StreamingGuide/0.1"})
    try:
        with urllib.request.urlopen(req, timeout=10) as resp:
            data = json.loads(resp.read())
            us = data.get("results", {}).get("US", {})
            flatrate = [p["provider_name"] for p in us.get("flatrate", [])]
            result = {"flatrate": flatrate}
            cache_set(db, key, result, ttl=86400)
            return result
    except Exception as e:
        print(f"TMDB provider error: {e}")
        return {}
```

## Task 6: Grid rendering — JavaScript

**Objective:** Fetch schedule data and render it in the grid

**Files:**
- Modify: `~/agentos/streaming-guide/static/app.js`

**Step 1:** Write the grid renderer

```javascript
// State
let currentShows = [];
let timeOffset = 0; // hours from now

// DOM refs
const guide = document.getElementById('guide');
const detail = document.getElementById('show-detail');
const searchInput = document.getElementById('search');
const timeLabel = document.getElementById('time-label');

// Fetch schedule
async function loadSchedule() {
  guide.innerHTML = '<div class="guide-loading">Loading listings...</div>';
  try {
    const resp = await fetch('/api/schedule');
    currentShows = await resp.json();
    renderGrid();
  } catch (e) {
    guide.innerHTML = '<div class="guide-loading">Failed to load. Check connection.</div>';
  }
}

// Build grid from shows
function renderGrid() {
  // Group shows by network/channel
  const channels = {};
  for (const show of currentShows) {
    const ch = show.network || 'Unknown';
    if (!channels[ch]) channels[ch] = [];
    channels[ch].push(show);
  }

  const channelNames = Object.keys(channels).sort();
  const startHour = new Date().getHours() + timeOffset;
  const hours = Array.from({length: 8}, (_, i) => (startHour + i) % 24);

  // Build HTML
  let html = '<div class="guide-grid">';

  // Header row
  html += '<div class="time-header"></div>'; // corner
  for (const h of hours) {
    const label = `${h % 12 || 12}:00${h < 12 ? 'a' : 'p'}`;
    html += `<div class="time-header">${label}</div>`;
  }

  // Channel rows
  for (const ch of channelNames) {
    html += `<div class="channel-label">${ch}</div>`;
    for (let i = 0; i < 8; i++) {
      html += '<div class="show-cell"></div>';
    }
  }

  html += '</div>';
  guide.innerHTML = html;

  // Now fill in shows
  // For now, just show title in first matching cell
  // Full time-slot matching comes in Task 7
  const cells = guide.querySelectorAll('.show-cell');
  let idx = 0;
  for (const ch of channelNames) {
    for (const show of channels[ch].slice(0, 8)) {
      if (idx < cells.length) {
        cells[idx].innerHTML = `
          <div class="show-title">${escapeHtml(show.name)}</div>
          <div class="show-episode">${escapeHtml(show.title || '')}</div>
        `;
        cells[idx].addEventListener('click', () => showDetail(show));
        idx++;
      }
    }
    // Skip remaining empty cells for this channel
    idx += 8 - Math.min(channels[ch].length, 8);
  }

  updateTimeLabel();
}

function updateTimeLabel() {
  const h = new Date().getHours() + timeOffset;
  timeLabel.textContent = `${h % 12 || 12}:00${h < 12 ? 'a' : 'p'}`;
}

function showDetail(show) {
  detail.innerHTML = `
    <button class="detail-close" onclick="this.parentElement.classList.add('hidden')">&times;</button>
    <h2>${escapeHtml(show.name)}</h2>
    <p><strong>${escapeHtml(show.network || '')}</strong> — ${escapeHtml(show.time || '')}</p>
    ${show.title ? `<p>Episode: ${escapeHtml(show.title)} (S${show.season}E${show.episode})</p>` : ''}
    ${show.summary ? `<p class="summary">${escapeHtml(show.summary)}</p>` : ''}
  `;
  detail.classList.remove('hidden');
}

function escapeHtml(str) {
  const div = document.createElement('div');
  div.textContent = str;
  return div.innerHTML;
}

// Time navigation
document.getElementById('prev-time').addEventListener('click', () => {
  timeOffset = Math.max(-12, timeOffset - 2);
  renderGrid();
});
document.getElementById('next-time').addEventListener('click', () => {
  timeOffset = Math.min(12, timeOffset + 2);
  renderGrid();
});

// Search
searchInput.addEventListener('input', async (e) => {
  const q = e.target.value.trim();
  if (q.length < 2) { loadSchedule(); return; }
  
  guide.innerHTML = '<div class="guide-loading">Searching...</div>';
  try {
    const resp = await fetch(`/api/search?q=${encodeURIComponent(q)}`);
    const results = await resp.json();
    if (results.length === 0) {
      guide.innerHTML = '<div class="guide-loading">No results found.</div>';
      return;
    }
    // Show search results in a simple list
    let html = '<div style="padding:16px;">';
    for (const r of results) {
      html += `
        <div style="padding:12px;border-bottom:1px solid var(--border);cursor:pointer;"
             onclick="this.querySelector('.summary').classList.toggle('hidden')">
          <strong>${escapeHtml(r.name)}</strong>
          ${r.year ? ` (${r.year})` : ''}
          <p class="summary hidden" style="margin-top:4px;color:var(--text-muted);font-size:12px">
            ${escapeHtml(r.overview || '')}
          </p>
        </div>`;
    }
    html += '</div>';
    guide.innerHTML = html;
  } catch (e) {
    guide.innerHTML = '<div class="guide-loading">Search failed.</div>';
  }
});

// Initial load
loadSchedule();
```

```javascript
// In renderGrid(), after building channel rows:
// Parse airtime to map shows to correct column
function getHourSlot(airtime) {
  if (!airtime) return -1;
  const [h, m] = airtime.split(':').map(Number);
  return h + (m >= 30 ? 0.5 : 0);
}

// Fill shows in correct time slots
for (const ch of channelNames) {
  const shows = channels[ch];
  for (const show of shows) {
    const showHour = getHourSlot(show.time);
    if (showHour < 0) continue;
    const colOffset = showHour - startHour;
    if (colOffset < 0 || colOffset >= 8) continue;
    
    // Calculate cell index
    const chIdx = channelNames.indexOf(ch);
    const cellIdx = chIdx * 8 + Math.floor(colOffset);
    
    if (cellIdx < cells.length) {
      cells[cellIdx].innerHTML = `
        <div class="show-title">${escapeHtml(show.name)}</div>
        ${show.title ? `<div class="show-episode">${escapeHtml(show.title)}</div>` : ''}
        ${renderBadges(show.streaming || [])}
      `;
      cells[cellIdx].addEventListener('click', () => showDetail(show));
    }
  }
}

function renderBadges(services) {
  if (!services || services.length === 0) return '';
  const slugs = {
    'Netflix': 'netflix', 'Amazon Prime Video': 'prime',
    'Hulu': 'hulu', 'Disney Plus': 'disney', 'Max': 'max',
    'Apple TV Plus': 'apple', 'Paramount Plus': 'paramount',
    'Peacock': 'peacock', 'Peacock Premium': 'peacock',
  };
  return '<div class="streaming-badges">' +
    services.slice(0, 3).map(s => {
      const cls = slugs[s] || '';
      return `<span class="badge ${cls}">${s}</span>`;
    }).join('') +
    '</div>';
}
```
